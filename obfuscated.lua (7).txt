-- Gerenciador de dados
local DataManager = {}
DataManager.__index = DataManager
function DataManager.new()
    local self = setmetatable({}, DataManager)
    self:_ensureFolder()
    return self
end
function DataManager:_ensureFolder()
    if not isfolder(CONFIG.FOLDER_NAME) then
        makefolder(CONFIG.FOLDER_NAME)
    end
end
function DataManager:save(fileName, data)
    local success, result = pcall(function()
        local jsonData = HttpService:JSONEncode(data)
        writefile(CONFIG.FOLDER_NAME .. "\x2f" .. fileName, jsonData)
    end)
    
    if not success then
        warn("\x45\x72\x72\x6f\x20\x61\x6f\x20\x73\x61\x6c\x76\x61\x72\x20\x64\x61\x64\x6f\x73\x3a", result)
    end
    return success
end
function DataManager:load(fileName)
    local filePath = CONFIG.FOLDER_NAME .. "\x2f" .. fileName
    
    if not isfile(filePath) then
        return nil
    end
    local success, result = pcall(function()
        local jsonData = readfile(filePath)
        return HttpService:JSONDecode(jsonData)
    end)
    
    if not success then
        warn("\x45\x72\x72\x6f\x20\x61\x6f\x20\x63\x61\x72\x72\x65\x67\x61\x72\x20\x64\x61\x64\x6f\x73\x3a", result)
        return nil
    end
    
    return result
end

local AuthManager = {}
AuthManager.__index = AuthManager
function AuthManager.new(dataManager)
    local self = setmetatable({}, AuthManager)
    self.dataManager = dataManager
    self.currentKey = nil
    return self
end
function AuthManager:isLinkValid()
    local savedData = self.dataManager:load(CONFIG.FILES.LINK)
    if not savedData or not savedData.time then
        return false
    end
    
    return (tick() - savedData.time) <= CONFIG.LINK_EXPIRY_TIME
end
function AuthManager:checkPremium()
    local premiumData = self.dataManager:load(CONFIG.FILES.PREMIUM)
    return premiumData and premiumData.key == CONFIG.PREMIUM_KEY
end
function AuthManager:isPremiumKey(inputKey)
    return inputKey == CONFIG.PREMIUM_KEY
end
function AuthManager:validateKey(inputKey)
    if not inputKey or inputKey == "" then
        return false, "\x41\x20\x73\x65\x6e\x68\x61\x20\x65\x20\x6f\x62\x72\x69\x67\x61\x74\xf3\x72\x69\x61\x2e"
    end
-- Verifica se √© a senha premium primeiro
    if self:isPremiumKey(inputKey) then
        return true, "\x50\x72\x65\x6d\x69\x75\x6d"
    end
-- Verifica senha normal
    local savedData = self.dataManager:load(CONFIG.FILES.LINK)
    if not savedData or not savedData.link then
        return false, "\x4c\x69\x6e\x6b\x20\x6e\xe3\x6f\x20\x65\x6e\x63\x6f\x6e\x74\x72\x61\x64\x6f\x20\x6f\x75\x20\x65\x78\x70\x69\x72\x61\x64\x6f\x2e"
    end
    
    local expectedKey = LINK_KEYS[savedData.link]
    if not expectedKey then
        return false, "\x4c\x69\x6e\x6b\x20\x69\x6e\x76\xe1\x6c\x69\x64\x6f\x2e"
    end
    
    return inputKey == expectedKey, inputKey == expectedKey and "\x4e\x6f\x72\x6d\x61\x6c" or "\x53\x65\x6e\x68\x61\x20\x6e\xe3\x6f\x20\x72\x65\x63\x6f\x6e\x68\x65\x63\x69\x64\x61\x2e"
end
function AuthManager:authenticate()
-- Verifica premium primeiro
    if self:checkPremium() then
        return true, "\x50\x72\x65\x6d\x69\x75\x6d", "\x41\x73\x73\x69\x6e\x61\x6e\x74\x65\x20\x50\x72\x65\x6d\x69\x75\x6d\x2c\x20\x49\x6e\x69\x63\x69\x61\x6c\x69\x7a\x61\x6e\x64\x6f\x20\x69\x6e\x74\x65\x72\x66\x61\x63\x65\x2e"
    end
-- Verifica autentica√ß√£o normal
    local savedData = self.dataManager:load(CONFIG.FILES.LINK)
    local savedKeyData = self.dataManager:load(CONFIG.FILES.KEY)
    
    if savedData and savedKeyData and self:isLinkValid() then
        local isValid, keyType = self:validateKey(savedKeyData.key)
        if isValid then
            return true, keyType, "\x53\x65\x6e\x68\x61\x20\x72\x65\x63\x6f\x6e\x68\x65\x63\x69\x64\x61\x2c\x20\x49\x6e\x69\x63\x69\x61\x6c\x69\x7a\x61\x6e\x64\x6f\x20\x69\x6e\x74\x65\x72\x66\x61\x63\x65\x2e"
        end
    end
    
    return false, "\x46\x61\x6c\x68\x61", "\x41\x75\x74\x65\x6e\x74\x69\x63\x61\xe7\xe3\x6f\x20\x6e\x65\x63\x65\x73\x73\xe1\x72\x69\x61\x2e"
end
function AuthManager:generateRandomLink()
    local links = {}
    for link in pairs(LINK_KEYS) do
        table.insert(links, link)
    end
    
    if #links == 0 then
        return nil
    end
    
    return links[math.random(#links)]
end

		return obj.Position
	end

-- Verifica se h√° jogador pr√≥ximo com Tool equipada
local function isPlayerWithToolNearby()
    local myChar = LocalPlayer.Character
    if not myChar then return false end

    local myRoot = myChar:FindFirstChild("\x48\x75\x6d\x61\x6e\x6f\x69\x64\x52\x6f\x6f\x74\x50\x61\x72\x74")
    if not myRoot then return false end

    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local root = player.Character:FindFirstChild("\x48\x75\x6d\x61\x6e\x6f\x69\x64\x52\x6f\x6f\x74\x50\x61\x72\x74")
            local tool = player.Character:FindFirstChildWhichIsA("\x54\x6f\x6f\x6c")
            if root and tool then
                local distance = (myRoot.Position - root.Position).Magnitude
                if distance <= 30 then
                    return true
                end
            end
        end
    end

    return false
end

    if not plotsFolder then return end

    for _, plot in ipairs(plotsFolder:GetChildren()) do

-- Update all Esp elements
local function updateBrainrotEsp()
    local plots = workspace:FindFirstChild("\x50\x6c\x6f\x74\x73")
    if not plots then return end
    
    for _, plot in ipairs(plots:GetChildren()) do
        local podiums = plot:FindFirstChild("\x41\x6e\x69\x6d\x61\x6c\x50\x6f\x64\x69\x75\x6d\x73")
        if podiums then
            for _, podium in ipairs(podiums:GetChildren()) do
                local displayName, rarity = getOverheadAndRarity(podium)
                local settings = raritySettings[rarity]
                
                if displayName and rarity and settings and settings.Enabled then
                    if not brainrotEsp[podium] then
                        createBrainrotEsp(podium, displayName, rarity, settings)
                    end
                else
                    removeBrainrotEsp(podium)
                end
            end
        end
    end
    
    -- Update distances for existing Esp
    for podium, data in pairs(brainrotEsp) do
        if not podium.Parent then
            removeBrainrotEsp(podium)
        elseif data.displayName and data.infoLabel then
            local part = podium.PrimaryPart or podium:FindFirstChildWhichIsA("\x42\x61\x73\x65\x50\x61\x72\x74")
            if part and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("\x48\x75\x6d\x61\x6e\x6f\x69\x64\x52\x6f\x6f\x74\x50\x61\x72\x74") then
                local dist = (LocalPlayer.Character.HumanoidRootPart.Position - part.Position).Magnitude
                data.infoLabel.Text = string.format("\x5b\x25\x73\x5d\x20\x44\x69\x73\x74\x61\x6e\x63\x65\x3a\x20\x25\x2e\x30\x66", data.rarity, dist)
                
                -- Color distance based on proximity (closer = more intense)
                local intensity = math.max(0.3, 1 - (dist / 200))
                data.infoLabel.TextTransparency = 1 - intensity
            end
        end
    end
end

-- Start Esp system
local function StartBrainrotEsp()
    if brainrotConnection then return end
    brainrotConnection = RunService.Heartbeat:Connect(updateBrainrotEsp)
end

-- Stop Esp system
local function StopBrainrotEsp()
    if brainrotConnection then
        brainrotConnection:Disconnect()
        brainrotConnection = nil
    end
    for podium in pairs(brainrotEsp) do
        removeBrainrotEsp(podium)
    end
end

-- Check if any Esp is enabled
local function isAnyEspEnabled()
    for _, settings in pairs(raritySettings) do
        if settings.Enabled then
            return true
        end
    end
    return false
end

    return EspTab:CreateToggle({

    })

        function serverHop()
            local cursor = nil
            local PlaceId, JobId = game.PlaceId, game.JobId

            hops += 1
            if hops >= maxHopsBeforeReset then

                hops = 0

            while tries < 5 do

                    return HttpService:JSONDecode(game:HttpGet(url))
                end)

                if success and response and response.data then
                    local servers = {}
                    for _, server in ipairs(response.data) do
                        if tonumber(server.playing or 0) < tonumber(server.maxPlayers or 1)
                            and server.id ~= JobId
                            and not visitedJobIds[server.id]
                        then
                            table.insert(servers, server.id)
                        end
                    end

                    if #servers > 0 then
                        local picked = servers[math.random(1, #servers)]
                        teleportFails = 0
                        TeleportService:TeleportToPlaceInstance(PlaceId, picked)
                        return
                    end

                    cursor = response.nextPageCursor
                    if not cursor then
                        tries += 1
                        wait(2)
                    end
                else
                    tries += 1
                    wait(2)
                end
            end

            TeleportService:Teleport(PlaceId)
        end

    function RTXSystem:createReflectionSystem()
        local materials = {
            Enum.Material.Glass,
            Enum.Material.Mirror,
            Enum.Material.Metal,
            Enum.Material.Neon

        local function createReflection(part)
            local gui = Instance.new("\x53\x75\x72\x66\x61\x63\x65\x47\x75\x69", part)
            gui.Face = Enum.NormalId.Top
            local frame = Instance.new("\x46\x72\x61\x6d\x65", gui)

            frame.BackgroundColor3 = Color3.fromRGB(200, 220, 255)
            frame.BackgroundTransparency = 0.3
            local grad = Instance.new("\x55\x49\x47\x72\x61\x64\x69\x65\x6e\x74", frame)
            grad.Transparency = NumberSequence.new({
                NumberSequenceKeypoint.new(0, 0.8),
                NumberSequenceKeypoint.new(0.5, 0.3),
                NumberSequenceKeypoint.new(1, 0.8),

            local tween = TweenService:Create(grad, TweenInfo.new(2, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true), {Rotation = 45})

-- ‚ñº Cor por rank ou time
local function getPlayerRankColor(player)
    if player.Team then return player.TeamColor.Color end
    local name = player.DisplayName:lower()
    if name:find("\x61\x64\x6d\x69\x6e") then return Color3.fromRGB(255, 0, 0) end
    if name:find("\x76\x69\x70") then return Color3.fromRGB(255, 215, 0) end
    if name:find("\x6d\x6f\x64") then return Color3.fromRGB(0, 255, 255) end
    return Color3.fromRGB(255, 255, 255)
end

-- ‚ñº Cria√ß√£o do ESP
local function createESP(player)
    if player == LocalPlayer or ESPs[player] then return end

    -- Billboard (Texto)
    local billboard = Instance.new("\x42\x69\x6c\x6c\x62\x6f\x61\x72\x64\x47\x75\x69")
    billboard.Name = "\x45\x53\x50\x5f" .. player.Name
    billboard.AlwaysOnTop = true
    billboard.Size = UDim2.new(0, 300, 0, 30)
    billboard.StudsOffset = Vector3.new(0, 3, 0)

    local label = Instance.new("\x54\x65\x78\x74\x4c\x61\x62\x65\x6c")
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundTransparency = 1
    label.TextScaled = true
    label.Font = Enum.Font.SourceSansBold
    label.TextColor3 = Color3.new(1, 1, 1)
    label.TextStrokeTransparency = 0
    label.Text = "\x45\x53\x50"
    label.Parent = billboard

    ESPs[player] = {
        Billboard = billboard,
        Label = label
    }

    -- Highlight
    local highlight = Instance.new("\x48\x69\x67\x68\x6c\x69\x67\x68\x74")
    highlight.Name = player.Name
    highlight.FillColor = getPlayerRankColor(player)
    highlight.FillTransparency = 0.5
    highlight.OutlineColor = Color3.new(1, 1, 1)
    highlight.OutlineTransparency = 0
    highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    highlight.Parent = HighlightFolder
    Highlights[player] = highlight

    -- Aponta para personagem atual
    if player.Character and player.Character:FindFirstChild("\x48\x75\x6d\x61\x6e\x6f\x69\x64\x52\x6f\x6f\x74\x50\x61\x72\x74") then
        highlight.Adornee = player.Character
        billboard.Parent = player.Character.HumanoidRootPart
    end

    -- Atualiza ao renascer
    Connections[player] = player.CharacterAdded:Connect(function(char)
        task.wait(1)
        highlight.Adornee = char
        local hrp = char:FindFirstChild("\x48\x75\x6d\x61\x6e\x6f\x69\x64\x52\x6f\x6f\x74\x50\x61\x72\x74")
        if hrp then
            billboard.Parent = hrp
        end
    end)
end

-- ‚ñº Atualiza ESP (texto, dist√¢ncia, vida)
local function updateESP(player)
    local esp = ESPs[player]
    if not esp then return end
    local character = player.Character
    if not character then return end

    local root = character:FindFirstChild("\x48\x75\x6d\x61\x6e\x6f\x69\x64\x52\x6f\x6f\x74\x50\x61\x72\x74")
    local humanoid = character:FindFirstChild("\x48\x75\x6d\x61\x6e\x6f\x69\x64")
    if not root or not humanoid then return end

    local dist = (LocalPlayer.Character.HumanoidRootPart.Position - root.Position).Magnitude

    -- Esconde se estiver longe
    esp.Billboard.Enabled = getgenv().ShowBillboardESP and dist <= MAX_DISTANCE
    Highlights[player].Enabled = getgenv().ShowHighlightESP and dist <= MAX_DISTANCE

    -- Texto
    local name = player.DisplayName ~= "" and player.DisplayName or player.Name
    local health = math.floor(humanoid.Health)
    local text = string.format("\x4e\x6f\x6d\x65\x3a\x20\x25\x73\x20\x7c\x20\x56\x69\x64\x61\x3a\x20\x25\x64\x20\x7c\x20\x44\x69\x73\x74\xe2\x6e\x63\x69\x61\x3a\x20\x25\x64", name, health, math.floor(dist))
    esp.Label.Text = text
end

-- ‚ñº Remove ESP
local function removeESP(player)
    if ESPs[player] then
        if ESPs[player].Billboard then ESPs[player].Billboard:Destroy() end
        ESPs[player] = nil
    end
    if Highlights[player] then
        Highlights[player]:Destroy()
        Highlights[player] = nil
    end
    if Connections[player] then
        Connections[player]:Disconnect()
        Connections[player] = nil
    end
end
